    def generate_agent_table(self) -> Table:
        """Generate table of active agents"""
        table = Table(title="Active Agents")
        table.add_column("Agent")
        table.add_column("Status")
        table.add_column("Task")

        for model in ["codellama:13b", "qwq", "mistral", "codegeex4"]:
            status = "active" if self.active_agents.get(model, {}).get("status") == "active" else "offline"

            table.add_row(model, status, task, progress, style=style)

    def generate_pattern_panel(self) -> Panel:
        """Generate panel of detected patterns"""
        # Get real file system patterns
        fs_patterns = []
        try:
            workspace = Path.cwd()

            # Track file changes
            total_files = len(list(workspace.rglob("*")))
            python_files = len(list(workspace.rglob("*.py")))
            yaml_files = len(list(workspace.rglob("*.yml")))

            fs_patterns.extend([
                f"Total files: {total_files}",
                f"Python files: {python_files}",
                f"YAML configs: {yaml_files}"
            ])

            # Track recent changes
            recent_changes = []
            for f in workspace.rglob("*"):
                if f.is_file():
                    mtime = datetime.fromtimestamp(f.stat().st_mtime)
                    if (datetime.now() - mtime).total_seconds() < 300:  # Changed in last 5 min
                        recent_changes.append(f.relative_to(workspace))

            if recent_changes:
                fs_patterns.append("\nRecent changes:")
                for change in recent_changes[:5]:
                    fs_patterns.append(f"- {change}")

        except Exception as e:
            fs_patterns.append(f"Error scanning files: {e}")

        return Panel("\n".join(fs_patterns), title="Emerging Patterns")

    def generate_metrics_panel(self) -> Panel:
        """Generate panel of system metrics"""
        try:
            cpu = psutil.cpu_percent(interval=0.1)
            mem = psutil.virtual_memory()
            disk = psutil.disk_usage('/')

            # Get active model info
            active_models = [
                model for model, info in self.active_agents.items()
                if info.get("status") == "active"
            ]

            metrics = [
                f"CPU: {cpu}%",
                f"Memory: {mem.percent}%",
                f"Disk: {disk.percent}%",
                "",
                "Active Models:",
            ]

            for model in active_models:
                metrics.append(f"- {model}")

            # Add process info
            python_procs = []
            for proc in psutil.process_iter(['name', 'cpu_percent', 'memory_percent']):
                try:
                    if 'python' in proc.info['name'].lower():
                        python_procs.append(proc)
                except (psutil.NoSuchProcess, psutil.AccessDenied):
                    pass

            if python_procs:
                metrics.extend([
                    "",
                    "Python Processes:",
                ])
                for proc in sorted(python_procs, key=lambda p: p.info['cpu_percent'], reverse=True)[:3]:
                    metrics.append(f"- {proc.info['name']}: {proc.info['cpu_percent']:.1f}% CPU")

            return Panel("\n".join(metrics), title="System Metrics")

        except Exception as e:
            return Panel(f"Error getting metrics: {e}", title="System Metrics")

    def update_display(self):

    async def organize_by_type(self):
        """Organize files by their extension"""
        try:
            # Create organization directories
            categories = {
                'python': ['.py'],
                'docs': ['.md', '.txt', '.pdf', '.doc', '.docx'],
                'data': ['.json', '.yml', '.yaml', '.csv'],
                'configs': ['.env', '.ini', '.cfg', '.conf'],
                'scripts': ['.sh', '.bash', '.zsh'],
                'tests': ['_test.py', 'test_', '_spec.py'],
                'backups': ['.bak', '.backup', '.old']
            }

            for category in categories:
                os.makedirs(category, exist_ok=True)

            # Track stats
            moved = 0
            organized = {}

            # Move files to appropriate directories
            for file in Path('.').rglob('*'):
                if file.is_file():
                    for category, extensions in categories.items():
                        if any(str(file).endswith(ext) for ext in extensions):
                            if category not in organized:
                                organized[category] = 0
                            try:
                                target = Path(category) / file.name
                                shutil.move(str(file), str(target))
                                organized[category] += 1
                                moved += 1
                            except Exception as e:
                                self.console.print(f"[red]Error moving {file}: {e}")

            # Show results
            self.console.print("\n[green]Files organized:")
            for category, count in organized.items():
                self.console.print(f"[cyan]{category}:[/cyan] {count} files")
            self.console.print(f"\n[green]Total files moved: {moved}")

        except Exception as e:
            self.console.print(f"[red]Error organizing files: {e}")

    async def cleanup_temp(self):
        """Clean up temporary and cache files"""
        try:
            patterns = ['*.pyc', '*.pyo', '*.pyd', '*.so', '*.o', '*.cache',
                       '*__pycache__*', '*.egg-info', '*.log', '.DS_Store']
            removed = 0

            for pattern in patterns:
                for file in Path('.').rglob(pattern):
                    try:
                        if file.is_file():
                            file.unlink()
                            removed += 1
                        elif file.is_dir():
                            shutil.rmtree(str(file))
                            removed += 1
                    except Exception as e:
                        self.console.print(f"[red]Error removing {file}: {e}")

            self.console.print(f"[green]Cleaned up {removed} temporary files/directories")

        except Exception as e:
            self.console.print(f"[red]Error during cleanup: {e}")

    async def show_stats(self):
        """Show detailed file statistics"""
        try:
            stats = {
                'extensions': {},
                'largest_files': [],
                'newest_files': [],
                'total_size': 0,
                'total_files': 0,
                'total_dirs': 0
            }

            # Collect stats
            for file in Path('.').rglob('*'):
                if file.is_file():
                    # Count extensions
                    ext = file.suffix
                    stats['extensions'][ext] = stats['extensions'].get(ext, 0) + 1

                    # Track file size
                    size = file.stat().st_size
                    stats['total_size'] += size
                    stats['total_files'] += 1

                    # Track largest files
                    stats['largest_files'].append((file, size))

                    # Track newest files
                    mtime = file.stat().st_mtime
                    stats['newest_files'].append((file, mtime))

                elif file.is_dir():
                    stats['total_dirs'] += 1

            # Sort and limit lists
            stats['largest_files'].sort(key=lambda x: x[1], reverse=True)
            stats['largest_files'] = stats['largest_files'][:5]
            stats['newest_files'].sort(key=lambda x: x[1], reverse=True)
            stats['newest_files'] = stats['newest_files'][:5]

            # Display stats
            self.console.print("\n[bold cyan]File Statistics:[/bold cyan]")
            self.console.print(f"\nTotal Files: {stats['total_files']}")
            self.console.print(f"Total Directories: {stats['total_dirs']}")
            self.console.print(f"Total Size: {stats['total_size'] / 1024 / 1024:.2f} MB")

            self.console.print("\n[bold cyan]Extensions:[/bold cyan]")
            for ext, count in sorted(stats['extensions'].items(), key=lambda x: x[1], reverse=True):
                self.console.print(f"{ext or 'no ext'}: {count} files")

            self.console.print("\n[bold cyan]Largest Files:[/bold cyan]")
            for file, size in stats['largest_files']:
                self.console.print(f"{file}: {size / 1024 / 1024:.2f} MB")

            self.console.print("\n[bold cyan]Most Recent Files:[/bold cyan]")
            for file, mtime in stats['newest_files']:
                self.console.print(f"{file}: {datetime.fromtimestamp(mtime)}")

        except Exception as e:
            self.console.print(f"[red]Error getting stats: {e}")

    async def find_duplicates(self):
        """Find duplicate files based on content hash"""
        try:
            duplicates = {}
            processed = 0

            self.console.print("[yellow]Scanning for duplicates (this may take a while)...")

            for file in Path('.').rglob('*'):
                if file.is_file():
                    processed += 1
                    if processed % 100 == 0:
                        self.console.print(f"Processed {processed} files...")

                    # Calculate file hash
                    file_hash = self.get_file_hash(file)
                    if file_hash:
                        if file_hash in duplicates:
                            duplicates[file_hash].append(file)
                        else:
                            duplicates[file_hash] = [file]

            # Show only actual duplicates
            real_dupes = {k: v for k, v in duplicates.items() if len(v) > 1}

            if real_dupes:
                self.console.print("\n[bold red]Found Duplicate Files:[/bold red]")
                for hash_val, files in real_dupes.items():
                    self.console.print(f"\n[cyan]Hash: {hash_val}[/cyan]")
                    for file in files:
                        self.console.print(f"  {file}")
            else:
                self.console.print("[green]No duplicate files found!")

        except Exception as e:
            self.console.print(f"[red]Error finding duplicates: {e}")

    def get_file_hash(self, file_path: Path, block_size: int = 65536) -> str:
        """Calculate SHA256 hash of a file"""
        import hashlib
        try:
            sha256 = hashlib.sha256()
            with open(file_path, 'rb') as f:
                for block in iter(lambda: f.read(block_size), b''):
                    sha256.update(block)
            return sha256.hexdigest()
        except Exception as e:
            self.console.print(f"[red]Error hashing {file_path}: {e}")
            return None

    async def monitor_patterns(self):
        """Monitor real file system patterns"""
        last_scan = {}
        while self.running:
            try:
                workspace = Path.cwd()
                current_scan = {}

                # Scan workspace
                for f in workspace.rglob("*"):
                    if f.is_file():
                        try:
                            current_scan[f] = {
                                "mtime": f.stat().st_mtime,
                                "size": f.stat().st_size
                            }
                        except Exception:
                            continue

                # Detect changes
                if last_scan:
                    changes = []
                    for f, stats in current_scan.items():
                        if f not in last_scan:
                            changes.append(f"New file: {f.relative_to(workspace)}")
                        elif stats != last_scan[f]:
                            changes.append(f"Modified: {f.relative_to(workspace)}")

                    for f in last_scan:
                        if f not in current_scan:
                            changes.append(f"Deleted: {f.relative_to(workspace)}")

                    if changes:
                        self.patterns.extend(changes)
                        # Keep last 100 patterns
                        self.patterns = self.patterns[-100:]

                last_scan = current_scan

            except Exception as e:
                logging.error(f"Pattern monitoring error: {e}")
            await asyncio.sleep(5)

            self.console.print(f"[red]Error processing patterns: {e}")

    def _analyze_workspace_structure(self) -> dict:
        """Analyze workspace directory structure"""
        structure = {}
        try:
            workspace = Path.cwd()
            for path in workspace.rglob("*"):
                if path.is_file():
                    relative = path.relative_to(workspace)
                    parts = relative.parts
                    current = structure
                    for part in parts[:-1]:
                        current = current.setdefault(part, {})
                    current[parts[-1]] = path.stat().st_size
        except Exception as e:
            logging.error(f"Structure analysis failed: {e}")
        return structure

    def _detect_code_patterns(self) -> list:
        """Detect patterns in code files"""
        patterns = []
        try:
            workspace = Path.cwd()
            for py_file in workspace.rglob("*.py"):
                with open(py_file) as f:
                    content = f.read()
                    # Add basic code patterns
                    if "class" in content:
                        patterns.append({"type": "class", "file": str(py_file)})
                    if "async" in content:
                        patterns.append({"type": "async", "file": str(py_file)})
                    if "import" in content:
                        patterns.append({"type": "import", "file": str(py_file)})
        except Exception as e:
            logging.error(f"Pattern detection failed: {e}")
        return patterns

    def _analyze_dependencies(self) -> dict:
        """Analyze project dependencies"""
        deps = {}
        try:
            # Check Python dependencies
            if Path("requirements.txt").exists():
                with open("requirements.txt") as f:
                    deps["python"] = f.read().splitlines()

            # Check package.json if it exists
            if Path("package.json").exists():
                with open("package.json") as f:
                    deps["node"] = json.load(f).get("dependencies", {})

        except Exception as e:
            logging.error(f"Dependency analysis failed: {e}")
        return deps

    async def start(self):
        """Start the monitoring system with AI integration"""
        # Initialize AI gateway
        await self.gateway.initialize()

        # Start background tasks
        agent_task = asyncio.create_task(self.monitor_agents())
        pattern_task = asyncio.create_task(self.monitor_patterns())
        input_task = asyncio.create_task(self.handle_input())

        try:
            with Live(self.layout, refresh_per_second=1, screen=False):
                while self.running:
                    self.update_display()
                    await asyncio.sleep(1)

        finally:
            # Cleanup tasks
            agent_task.cancel()
            pattern_task.cancel()
            input_task.cancel()
            try:
                await asyncio.gather(agent_task, pattern_task, input_task)
            except asyncio.CancelledError:
                pass
